CREATE DATABASE INVENTARIOS_SOFT
CREATE TABLE PROCESADOR (IDPROCESADOR int identity (1,1) Primary key , PROCESADOR VARCHAR(50), MODELO VARCHAR(50), VELOCIDAD VARCHAR(20),FECHA_CR DATE,HORA_CR TIME, OBS VARCHAR(200))
CREATE TABLE BOARD (IDBOARD int identity (1,1) Primary key ,MARCA VARCHAR(100), MODELO VARCHAR(50),FECHA_CR DATE,HORA_CR TIME, OBS VARCHAR(200))
CREATE TABLE F_PODER(IDF_PODER int identity (1,1) Primary key ,MARCA VARCHAR(100), MODELO VARCHAR(50),VOLTAJE VARCHAR(30),FECHA_CR DATE,HORA_CR TIME, OBS VARCHAR(200))
CREATE TABLE RAM(IDRAM int identity (1,1) Primary key ,MARCA VARCHAR(100), TIPO VARCHAR(20), CAPACIDAD VARCHAR(20), BUS VARCHAR(20),FECHA_CR DATE,HORA_CR TIME, OBS VARCHAR(200))
CREATE TABLE HDD(IDHDD int identity (1,1) Primary key , MARCA VARCHAR(30), CAPACIDAD VARCHAR(20), TIPO VARCHAR(20),FECHA_CR DATE,HORA_CR TIME, OBS VARCHAR (200))
CREATE TABLE UNIDAD_DVD(IDUNIDAD_DVD int identity (1,1) Primary key ,MARCA VARCHAR(30), MODELO VARCHAR(50), TIPO VARCHAR(20),FECHA_CR DATE,HORA_CR TIME, OBS VARCHAR(200))
CREATE TABLE ACCESORIOS_HW(IDACCESORIO int identity (1,1) Primary key , DESCRIPCION VARCHAR(150),MARCA VARCHAR(100), MODELO VARCHAR(50),TIPO VARCHAR(30),FECHA_CR DATE,HORA_CR TIME,  OBS VARCHAR(200))
CREATE TABLE PC_GENERICO(IDPC_GEN int identity (1,1) Primary key , BOARD INT FOREIGN KEY REFERENCES BOARD(IDBOARD), PROCESADOR INT FOREIGN KEY REFERENCES PROCESADOR(IDPROCESADOR), RAMSLOT1 INT FOREIGN KEY REFERENCES RAM(IDRAM),RAMSLOT2 INT FOREIGN KEY REFERENCES RAM(IDRAM),FUENTE_PODER INT FOREIGN KEY REFERENCES F_PODER(IDF_PODER),HDD INT FOREIGN KEY REFERENCES HDD(IDHDD),UNIDAD_DVD INT FOREIGN KEY REFERENCES UNIDAD_DVD(IDUNIDAD_DVD),ACCESORIO INT FOREIGN KEY REFERENCES ACCESORIOS_HW(IDACCESORIO),FECHA_CR DATE,HORA_CR TIME,OBS VARCHAR(200))
CREATE TABLE TORRE(IDTORRE int identity (1,1)Primary key,Descripcion Varchar(100),Telefono Varchar(20),Direccion Varchar(60),FECHA_CR DATE,HORA_CR TIME,OBS VARCHAR (200))
CREATE TABLE PC_SALONES(ID_SALON int identity (1,1) Primary key,IDTORRE INT FOREIGN KEY REFERENCES TORRE(IDTORRE), SALON VARCHAR(50), PROCESADOR VARCHAR(60),RAMTOTAL VARCHAR(10),HDD VARCHAR(20),CANT_PC VARCHAR(2),FECHA_CR DATE,HORA_CR TIME,OBS VARCHAR(200))
CREATE TABLE EQUIPOS_SALAS(N_EQUIPO int Primary key ,ACTIVO_PC VARCHAR(6) ,PC_GENERICO INT FOREIGN KEY REFERENCES PC_GENERICO(IDPC_GEN),SALONES INT FOREIGN KEY REFERENCES PC_SALONES(ID_SALON),FECHA_CR DATE,HORA_CR TIME, OBS VARCHAR(200))
CREATE TABLE PERISFERICO(IDPERISFERICO int identity (1,1) Primary key ,N_EQUIPO INT FOREIGN KEY REFERENCES EQUIPOS_SALAS(N_EQUIPO),ACTIVO VARCHAR(6), DESCRIPCION VARCHAR(150),MARCA VARCHAR(100), MODELO VARCHAR(50),TIPO VARCHAR(30),SERIAL VARCHAR(40),FECHA_CR DATE,HORA_CR TIME, OBS VARCHAR(200))
CREATE TABLE BITACORA(ID_BITACORA int identity (1,1) Primary key, N_EQUIPO INT FOREIGN KEY REFERENCES EQUIPOS_SALAS(N_EQUIPO), DESCRIPCION VARCHAR(250), FECHA DATE, HORA TIME )
CREATE TABLE LIST_SOFTWARE(ID_SOFT int identity (1,1) Primary key, NOMBRE VARCHAR(50),VERSION_S VARCHAR(20), TIPO VARCHAR(50), ID_SALON INT FOREIGN KEY REFERENCES PC_SALONES(ID_SALON),FECHA_CR DATE,HORA_CR TIME,OBS VARCHAR(200))
CREATE TABLE C_USUARIO(ID_USUARIO INT IDENTITY (1,1) PRIMARY KEY, NOMBRE_C VARCHAR(80),FECHA_NAC DATE,CEDULA VARCHAR (20),EDAD VARCHAR(2),CARGO VARCHAR(30),DEPARTAMENTO VARCHAR(30),TELEFONO VARCHAR(20),CELULAR VARCHAR(20),DIRECCION VARCHAR(80),EMAIL VARCHAR(100),N_USUARIO VARCHAR(20),PASSWOR1 VARCHAR(50),FECHAC DATE,HORAC TIME, ACTIVO VARCHAR(1),OBS VARCHAR(250))
CREATE TABLE C_ROL(IDROL INT IDENTITY (1,1) PRIMARY KEY,ROL VARCHAR(20),IDUSUARIO INT FOREIGN KEY REFERENCES C_USUARIO(ID_USUARIO), OBS VARCHAR(250))
